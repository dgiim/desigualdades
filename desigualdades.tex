\begin{center}

\subsection{Desigualdades en espacios vectoriales.}
\noindent Desigualdad de Cauchy-Schwarz.
\begin{center}
 $\displaystyle |{\langle a,b \rangle}| \leq \|a\|\|b\|$
\end{center}

\noindent Desigualdad triangular.
\begin{center}
 $\displaystyle \|x+y\| \leq \|x\|+\|y\|$
\end{center}
